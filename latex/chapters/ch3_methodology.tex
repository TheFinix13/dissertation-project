\section{Probabilistic Forecaster}
% Two-layer LSTM, Gaussian NLL loss, uncertainty score $u_t$.
\begin{equation}\label{eq:gaussian_nll}
  \mathcal{L}_{\text{NLL}}
  = \frac{1}{N}\sum_{i=1}^{N}
    \left[\frac{(y_i - \mu_i)^2}{2\sigma_i^2}
    + \frac{1}{2}\ln\sigma_i^2\right]
\end{equation}

\section{Evaluation Metrics}
\begin{equation}\label{eq:sharpe}
  \text{Sharpe} = \frac{\bar{r}_p - r_f}{\sigma_p}
\end{equation}
\begin{equation}\label{eq:mdd}
  \text{MDD} = \max_{t \in [0,T]}
    \frac{\max_{s \leq t} V_s - V_t}{\max_{s \leq t} V_s}
\end{equation}
